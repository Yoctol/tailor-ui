# DatePicker

To select or input a date.

## When To Use

By clicking the input box, you can select a date from a popup calendar.

## Examples

```js
import { DatePicker } from 'tailor-ui';
```

### Date picker

#### Basic usage

<Playground>
  <DatePicker defaultValue={moment()} onChange={console.log} />
</Playground>

#### Controlled usage

<Playground>
  {() => {
    const [value, setValue] = React.useState(() => moment());

    return (
      <>
        <DatePicker value={value} onChange={setValue} />
        <Button mt="2" onClick={() => setValue(moment().subtract(3, 'days'))}>Set to 3 days ago</Button>
      </>
    )

}}

</Playground>

#### Customize input

<Playground>
  <DatePicker
    defaultValue={moment()}
    onChange={console.log}
    inputProps={{
      size: 'sm',
      width: 150,
    }}
  />
</Playground>

#### Choose Time

<Playground>
  <DatePicker
    onChange={console.log}
    placeholder="Please Select Date & Time"
    showTime
    showSecond
  />
</Playground>

#### Choose Time & with minute step

<Playground>
  <DatePicker
    onChange={console.log}
    format="YYYY-MM-DD HH:mm"
    minuteStep={15}
    showTime
  />
</Playground>

#### Disabled Specified Date

<Playground>
  <DatePicker
    disabledDate={current => {
      const date = moment({ hour: 0, minute: 0, second: 0 });
      return current.isBefore(date);
    }}
    onChange={console.log}

/>

</Playground>
