# Mention

Mention component.

## When To Use

When need to mention someone or something.

## Examples

```js
import { Mention } from '@tailor-ui/lab';
```

### Basic

<Playground>
  {() => {
    const suggestions = ['飲料', '食物'];
    const [value, setValue] = React.useState(
      Mention.convertStringToEditorState('我想要一杯 @飲料', suggestions)
    );
    return (
      <Mention
        suggestions={suggestions}
        value={value}
        onChange={value => {
          setValue(value);
          console.log('Raw value: ', Mention.convertEditorStateToString(value));
        }}
      />
    );
  }}
</Playground>
