### With maxLength & error

<Playground>
  <Input>
    {({ value, bind }) => {
      const hasError = value === 'error';
      return (
        <TextField
          label="Please type 'error'"
          maxLength={12}
          error={hasError}
          message="Just show error"
          {...bind}
        />
      );
    }}
  </Input>
</Playground>
