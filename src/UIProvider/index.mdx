import { Playground } from 'docz';
import { Value, Toggle } from 'react-powerplug';

import { UIProvider, Modal, Button, Radio, Space, Flex, Popconfirm, DatePicker } from '../';
import * as locales from '../locale';

## Usage

`UIProvider` provides theme to components, and provides a uniform localization support for built-in text of components.

```js
import { UIProvider } from '@yoctol/ui';
import zh_TW from '@yoctol/ui/lib/locale/zh_TW';

...

<UIProvider locale={zh_TW}>
  {children}
</UIProvider>
```

## Examples

<Playground>
  <Value initial="en_US">
    {({ value, set }) => (
      <>
        <Radio.Group value={value} onChange={set}>
          <Radio value="en_US">English</Radio>
          <Radio value="zh_TW">中文</Radio>
        </Radio.Group>

        <Space mt="3">
          <UIProvider locale={locales[value]}>
            <>
              <Flex>
                <Toggle>
                  {({ on, toggle }) => (
                    <>
                      <Modal
                        title="This is a Modal"
                        visible={on}
                        closable
                        onConfirm={toggle}
                        onCancel={toggle}
                      >
                        <div>This is the content of Modal</div>
                      </Modal>
                      <Button onClick={toggle}>Open Modal</Button>
                    </>
                  )}
                </Toggle>

                <Button
                  ml="2"
                  onClick={() => {
                    Modal.confirm({
                      title: 'Do you Want to delete these items?',
                      content: 'Some descriptions',
                      onConfirm: () => console.log('confirmed!'),
                      onCancel: () => console.log('canceled!'),
                    })
                  }}
                >
                  Confirm Modal
                </Button>

                <Space ml="2">
                  <Popconfirm
                    placement="bottom"
                    content="Are you sure to delete this todo?"
                    onConfirm={() => console.log('confirm')}
                    onCancel={() => console.log('cancel')}
                  >
                    <Button>Popconfirm</Button>
                  </Popconfirm>
                </Space>
              </Flex>
              <Space mt="3">
                <DatePicker />
              </Space>
            </>
          </UIProvider>
        </Space>
      </>
    )}
  </Value>
</Playground>
