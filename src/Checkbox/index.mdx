## Usage

```js
import { FormField, Label, Checkbox, CheckboxField } from '@yoctol/ui';
```

### Checkbox

#### Basic

<Playground>
  <Toggle>
    {({ on, toggle }) => (
      <FormField>
        <Checkbox
          id="checkbox-basic"
          checked={on}
          onChange={toggle}
        />
        <Label htmlFor="checkbox-basic">Checkbox</Label>
      </FormField>
    )}
  </Toggle>
</Playground>

#### With disabled

<Playground>
  <FormField>
    <Checkbox id="checked-disabled" checked disabled />
    <Label htmlFor="checked-disabled">Checked Disabled</Label>
  </FormField>
  <FormField>
    <Checkbox id="disabled" disabled />
    <Label htmlFor="disabled">Disabled</Label>
  </FormField>
</Playground>

#### Customize color

<Playground>

### CheckboxField

#### Basic

<Playground>
  <Toggle>
    {({ on, toggle }) => (
      <CheckboxField checked={on} onChange={toggle}>
        Checkbox
      </CheckboxField>
    )}
  </Toggle>
</Playground>

#### With error

<Playground>
  <Toggle>
    {({ on, toggle }) => (
      <CheckboxField
        checked={on}
        onChange={toggle}
        error
        message="Something went wrong!"
      >
        Checkbox
      </CheckboxField>
    )}
  </Toggle>
</Playground>

### CheckboxField.Group

#### Basic

<Playground>
  <Value initial={['check_1']}>
    {({ set, value }) => (
      <CheckboxField.Group
        label="Checkbox Group"
        options={[
          { label: 'checkbox 1', value: 'check_1' },
          { label: 'checkbox 2', value: 'check_2' },
          { label: 'checkbox 3', value: 'check_3', disabled: true },
          { label: 'checkbox 4', value: 'check_4' },
        ]}
        values={value}
        onChange={set}
      />
    )}
  </Value>
</Playground>

#### With group & error

<Playground>
  <Value initial={[]}>
    {({ set, value }) => (
      <CheckboxField.Group
        label="Checkbox Group"
        error
        message="Something went wrong!"
        options={[
          { label: 'checkbox 1', value: 'check_1' },
          { label: 'checkbox 2', value: 'check_2' },
          { label: 'checkbox 3', value: 'check_3' },
          { label: 'checkbox 4', value: 'check_4' },
        ]}
        values={value}
        onChange={val => set(val)}
      />
    )}
  </Value>
</Playground>
